# hadoop

hadoop fs ls /

hadoop fs -copyFromLocal local_fpath hdfs_fpath  #copy file from local to hdfs
hadoop fs -copyToLocal hdfs_fpath local_fpath  #copy file from hdfs to local

hadoop fs -rmr hdfs_fpath  # delete a hdfs folder
